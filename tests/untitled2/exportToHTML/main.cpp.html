#include <iostream>

class A {
public:
    virtual void foo() { std::cout << "A::foo\n"; }
};

class B: public A {
public:
    void foo() { std::cout << "B::foo\n"; }
};

class C: public B {
public:
    void foo() { std::cout << "C::foo\n"; }
};

int main(){
    C c;
    B *pb = &c;
    pb->foo();
    return 0;
}
